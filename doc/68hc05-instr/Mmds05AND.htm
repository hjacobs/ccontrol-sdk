AND - Logical AND ;flags(N,Z)

AND #opr ; A <= A ^ opr ;(IMM) 8 bit immediate

AND opr ; A <= A ^ (opr) ;(DIR) 8 bit direct

AND opr ; A <= A ^ (opr) ;(EXT) 16 bit extended

AND ,X ; A <= A ^ (X) ;(IX) indexed no offset

AND opr,X ; A <= A ^ (opr+X) ;(IX1) indexed, 8 bit offset

AND opr,X ; A <= A ^ (opr+X) ;(IX2) indexed, 16 bit offset
