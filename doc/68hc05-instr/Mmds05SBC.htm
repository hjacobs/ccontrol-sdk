SBC - Subtract with Carry ;flags(N,Z,C)

SBC #opr ;A <= A - C - opr ;(IMM) 8 bit immediate

SBC opr ;A <= A - C - (opr) ;(DIR) 8 bit direct

SBC opr ;A <= A - C - (opr) ;(EXT) 16 bit extended

SBC ,X ;A <= A - C - (X) ;(IX) indexed no offset

SBC opr,X ;A <= A - C - (opr+X) ;(IX1) indexed, 8 bit offset

SBC opr,X ;A <= A - C - (opr+X) ;(IX2) indexed, 16 bit offset
