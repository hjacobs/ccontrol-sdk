ADD - Add without Carry into Acc ;flags(H,N,Z,C)

ADD #opr ;A <= A + opr ;(IMM) 8 bit immediate

ADD opr ;A <= A + (opr) ;(DIR) 8 bit direct

ADD opr ;A <= A + (opr) ;(EXT) 16 bit extended

ADD ,X ;A <= A + (X) ;(IX) indexed no offset

ADD opr,X ;A <= A + (opr+X) ;(IX1) indexed, 8 bit offset

ADD opr,X ;A <= A + (opr+X) ;(IX2) indexed, 16 bit offset
