ADC - Add with Carry into Acc ;flags(H,N,Z,C)

ADC #opr ; A <= A + C + opr ;(IMM) 8 bit immediate

ADC opr ; A <= A + C + (opr) ;(DIR) 8 bit direct

ADC opr ; A <= A + C + (opr) ;(EXT) 16 bit extended

ADC ,X ; A <= A + C + (X) ;(IX) indexed no offset

ADC opr,X ; A <= A + C + (opr+X) ;(IX1) indexed, 8 bit offset

ADC opr,X ; A <= A + C + (opr+X) ;(IX2) indexed, 16 bit offset
