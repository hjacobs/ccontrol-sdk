SUB - Subtract Memory from Accumulator ;flags(N,Z,C)

SUB #opr ;A <= A - opr ;(IMM) 8 bit immediate

SUB opr ;A <= A - (opr) ;(DIR) 8 bit direct

SUB opr ;A <= A - (opr) ;(EXT) 16 bit extended

SUB ,X ;A <= A - (X) ;(IX) indexed no offset

SUB opr,X ;A <= A - (opr+X) ;(IX1) indexed w/8 bit offset

SUB opr,X ;A <= A - (opr+X) ;(IX2) indexed w/16 bit offset
